main:   lui $r2 0xFF							d2ff
        ori $r2 $r2 0xFF						6aff
        ori $r0 $r0 6							6006
        jal fib									7007
        lui $r0 0								d000
        sw $r1 0($r0)							c100
self:   j self									8006

fib:
        addi $r2 $r2 -3							4afd
        sw $r0 0($r2)							c800
        sw $r1 1($r2)							c901
        sw $r3 2($r2)							cb02
        lui $r3 0								d300
        beq $r0 $r3 returnZero					a30e
        ori $r3 $r3 1							6f01
        beq $r0 $r3 returnOne					a309
        addi $r0 $r0 -1							40ff
        jal fib									7007
        sw $r1 1($r2)							c901
        lw $r0 0($r2)							e800
        addi $r0 $r0 -2							40fe
        jal fib									7007
        lw $r0 1($r2) # load old return value	e801
        add $r1 $r0 $r1							0146
        j return								801c

returnOne:
        lui $r1 0								d100
        ori $r1 $r1 1							6501
        j return								801c

returnZero:
        lui $r1 0								d100

return:
        lw $r0 0($r2)							e800
        lw $r3 2($r2)							eb02
        addi $r2 $r2 3							4a03
        jr $r3									9c00
